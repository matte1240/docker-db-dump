📝 ESEMPIO VELOCE:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

services:
  mysql_backup:
    image: db-backup-sidecar:latest
    environment:
      DB_TYPE: mysql
      DB_HOST: mysql
      DB_USER: root
      DB_PASSWORD: password
      CRON_SCHEDULE: "0 2 * * *"  # Ogni giorno alle 2 AM
    volumes:
      - ./backups:/backups
    command: ["schedule"]
